Tarea 13

Elaborar un programa en JAVA y Ruby que realice una busqueda de un numero en un vector, mostrando solamente un mensaje indicando cuantas veces apareció el numero. Ademas de imprimir el vector.

JAVA

/*
Tarea 13
Busqueda de un elemento en un vector
Estudiante: Erik Everardo Cavazos Hernández
Matrícula: 1811290
Clase: Lenguajes de Programación V4-V6 Jueves
Docente: Ismael Pimentel
*/
import javax.swing.JOptionPane;

public class tarea13{
    public static void main(String[] args){
        int n = Integer.parseInt(JOptionPane.showInputDialog(null,"Cantidad de elementos:"));
        int[] a = new int[n];
        int i;
        for(i=0;i<n;i++){
            a[i] = Integer.parseInt(JOptionPane.showInputDialog(null,"Introduzca un numero entero para la posicion [" + i +"]"));
        }
        int elementoBuscado = Integer.parseInt(JOptionPane.showInputDialog(null,"Elemento a buscar:"));
        int contador = 0;
        for(i=0;i<n;i++){
            if(a[i] == elementoBuscado){
                contador++;
            }
        }
        if(contador == 0){
            JOptionPane.showMessageDialog(null,"El numero " + elementoBuscado + " no fue encontrado en el vector");
        }
        else if(contador == 1){
            JOptionPane.showMessageDialog(null,"El numero " + elementoBuscado + " fue encontrado una vez en el vector");
        }
        else{
            JOptionPane.showMessageDialog(null,"El numero " + elementoBuscado + " fue encontrado " + contador + " veces en el vector");
        }
        String impresionArreglo = "A = {";
        for(i=0;i<n-1;i++){
            impresionArreglo += a[i] + ",";
        }
        impresionArreglo += a[n-1] + "}";
        JOptionPane.showMessageDialog(null,impresionArreglo);
    }
}
