Tarea 14

Construya un programa en JAVA y Ruby que, al recibir como datos un arreglo bidimensional de NxM elementos de tipo entero y un numero entero, determine cuántos veces se encuentra este numero dentro del arreglo. Además, que imprima la matriz al final

JAVA

/*
Tarea 14
Busqueda de un elemento en una matriz
Estudiante: Erik Everardo Cavazos Hernández
Matrícula: 1811290
Clase: Lenguajes de Programación V4-V6 Jueves
Docente: Ismael Pimentel
*/
import javax.swing.JOptionPane;
public class tarea14{
    public static void main(String[] args){
        int n = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de filas:"));
        int m = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de columnas:"));
        int[][] matriz = new int[n][m];
        int i,j,vecesEncontrado = 0;
        //leer y guardar datos en la matriz
        for(i = 0; i<n; i++){
            for(j = 0; j<m; j++){
                matriz[i][j] = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero para Matriz[" + (i+1) + "][" + (j+1) + "]"));
            }
        }
        int elementoBuscado = Integer.parseInt(JOptionPane.showInputDialog(null, "Escriba numero a buscar en la matriz:"));
        //buscar elemento en la matriz
        for(i = 0; i<n; i++){
            for(j = 0; j<m; j++){
                if(elementoBuscado == matriz[i][j]){
                    vecesEncontrado++;
                }
            }
        }
        //mostrar cuantas veces se encontró el elemento en la matriz
        switch(vecesEncontrado){
            case 0: JOptionPane.showMessageDialog(null,"El numero " + elementoBuscado + " no se encontro");break;
            case 1: JOptionPane.showMessageDialog(null,"El numero " + elementoBuscado + " se encontro 1 vez");break;
            default: JOptionPane.showMessageDialog(null,"El numero " + elementoBuscado + " se encontro " + vecesEncontrado + " veces");
        }
        //mostrar la matriz en pantalla
        String impresionMatriz = "";
        for(i = 0; i<n; i++){
            for(j = 0; j<m; j++){
                impresionMatriz += "(" + matriz[i][j] + ") ";
            }
            impresionMatriz += "\n";
        }
        impresionMatriz += "\nNota: Puede haber una visualizacion confusa de la matriz. Debe tomar en cuenta las filas y columnas";
        JOptionPane.showMessageDialog(null,impresionMatriz);
    }
}
