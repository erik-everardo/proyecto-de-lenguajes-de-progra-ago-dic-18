Tarea 15

Elaborar un programa en JAVA y Ruby que calcule la suma de dos matrices que ingrese el usuario (Nota: validar lo que sea ncesario)

JAVA

/*
Tarea 15
Suma de dos matrices que ingrese el usuario
Estudiante: Erik Everardo Cavazos Hernández
Matrícula: 1811290
Clase: Lenguajes de Programación V4-V6 Jueves
Docente: Ismael Pimentel
*/
import javax.swing.JOptionPane;
public class tarea15{
    public static void main(String[] args){
        int n1,n2,m1,m2,i,j;
        do{
            n1 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de filas para la primera matriz"));
            m1 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de columnas para la primera matriz"));
            n2 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de filas para la segunda matriz"));
            m2 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de columnas para la segunda matriz"));
            if(n1 != n2 || m2 != m1){
                JOptionPane.showMessageDialog(null,"No se pueden sumar las matrices. Deben ser del mismo tamanio");
            }
        }while(n1 != n2 || m2 != m1);
        int[][] matriz1 = new int[n1][m1];
        int[][] matriz2 = new int[n2][m2];
        int[][] matrizResultado = new int[n1][m1];
        //datos a la matriz 1
        JOptionPane.showMessageDialog(null,"Va a introducir los datos de la matriz 1");
        for(i = 0; i<n1; i++){
            for(j = 0; j<m1; j++){
                matriz1[i][j] = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese numero entero a matriz 1["+(i+1)+"]"));
            }
        }
        //datos a la matriz 2
        JOptionPane.showMessageDialog(null,"Va a introducir los datos de la matriz 2");
        for(i = 0; i<n2; i++){
            for(j = 0; j<m2; j++){
                matriz2[i][j] = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese numero entero a matriz 2["+(i+1)+"]"));
            }
        }
        //sumar cada elemento en la nueva matriz
        for(i=0; i<n1; i++){
            for(j = 0; j<m1; j++){
                matrizResultado[i][j] = matriz1[i][j] + matriz2[i][j];
            }
        }
        //imprimir la matriz resultante
        String impresionMatriz = "Matriz resultante\n";
        for(i = 0; i<n1; i++){
            for(j = 0; j<m1; j++){
                impresionMatriz += matrizResultado[i][j] + " ";
            }
            impresionMatriz += "\n";
        }
        JOptionPane.showMessageDialog(null,impresionMatriz);
    }
}
