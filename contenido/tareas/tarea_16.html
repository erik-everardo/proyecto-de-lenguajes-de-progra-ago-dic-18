Tarea 16

Elaborar un programa en JAVA y Ruby que calcule la multiplicacion de dos matrices que ingrese el usuario, (nota: validar lo que sea necesario)

JAVA

/*
Tarea 16
Multiplicacion de dos matrices que ingrese el usuario
Estudiante: Erik Everardo Cavazos Hernández
Matrícula: 1811290
Clase: Lenguajes de Programación V4-V6 Jueves
Docente: Ismael Pimentel

import javax.swing.JOptionPane;

public class tarea16{
    public static void main(String[] args){
        int i,j,k,n1,m1,n2,m2,salir=0;
        String[]opciones = {"Realizar otra multiplicacion","Salir"};
        
        while(salir==0){
            //asegurarse de que las matrices se pueden multiplicar. 
            //numero de filas de la primera matriz debe ser igual al numero de columnas de la segunda matriz
            do{
                n1 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de filas para la primera matriz"));
                m1 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de columnas para la primera matriz"));

                n2 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de filas para la segunda matriz"));
                m2 = Integer.parseInt(JOptionPane.showInputDialog(null,"Numero de columnas para la segunda matriz"));
                if(n1 != m2){
                    JOptionPane.showMessageDialog(null, "Numero de filas de la primera matriz debe ser igual al numero de columnas de la segunda matriz");
                }
            }while(n1 != m2);

            //declaracion de matrices
            int[][] matriz1 = new int[n1][m1];
            int[][] matriz2 = new int[n2][m2];
            int[][] matrizResultante = new int[n1][m2];

            //datos a la matriz 1
            JOptionPane.showMessageDialog(null,"Va a introducir los datos de la matriz 1");
            for(i = 0; i<n1; i++){
                for(j = 0; j<m1; j++){
                    matriz1[i][j] = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese numero entero a matriz 1["+(i+1)+"][" + (j+1)+"]"));
                }
            }

            //datos a la matriz 2
            JOptionPane.showMessageDialog(null,"Va a introducir los datos de la matriz 2");
            for(i = 0; i<n2; i++){
                for(j = 0; j<m2; j++){
                    matriz2[i][j] = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese numero entero a matriz 2["+(i+1)+"][" + (j+1)+"]"));
                }
            }

            //multiplicacion: se hace producto punto de fila por columna
            for(i = 0; i<n1; i++){
                for(j = 0; j<m2; j++){
                    for(k=0;k<m1;k++){
                        matrizResultante[i][j] += matriz1[i][k] * matriz2[k][j];
                    }
                }
            }

            //imprimir la matriz resultante
            String impresionMatriz = "Matriz resultante\n";
            for(i = 0; i<n1; i++){
                for(j = 0; j<m2; j++){
                    impresionMatriz += matrizResultante[i][j] + " ";
                }
                impresionMatriz += "\n";
            }
            JOptionPane.showMessageDialog(null,impresionMatriz);
            salir = JOptionPane.showOptionDialog(null,"Que desea hacer?", "Que desea hacer?", JOptionPane.YES_NO_CANCEL_OPTION,JOptionPane.INFORMATION_MESSAGE,null, opciones,opciones[0]);
        }
    }
}
